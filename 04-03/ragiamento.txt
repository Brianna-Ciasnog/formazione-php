$arr = [
    ['id' => 1, 'name' => 'a'],
    ['id' => 2, 'parent' => 1, 'name' => 'a-child'],
    ['id' => 3, 'parent' => 1, 'name' => 'a-child-2'],
    ['id' => 4, 'name' => 'c'],
];


$new = [];

foreach($arr as $key => $row) {
    if(!isset($row['parent'])) $new[$row['id']] = $row;
}

foreach($arr as $key => $child) {
    if(isset($child['parent'])) { 
        $new[$child['parent']]['child'][] = $child;
    }
}

print_r($new);
